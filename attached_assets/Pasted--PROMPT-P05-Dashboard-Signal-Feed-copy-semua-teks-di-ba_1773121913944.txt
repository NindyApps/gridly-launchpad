2. SIGNAL FEED — src/components/signals/SignalFeed.tsx:
Props: filters: FilterState
 
Fetch on mount:
  supabase.from('intent_signals')
    .select('*')
    .eq('workspace_id', workspace.id)
    .eq('dismissed', false)
    .order('confidence_score', { ascending: false })
    .limit(50)
 
REALTIME: Subscribe to Supabase Realtime on intent_signals:
  supabase.channel('signals').on('postgres_changes',
    { event: 'INSERT', schema: 'public', table: 'intent_signals',
      filter: `workspace_id=eq.${workspace.id}` },
    (payload) => {
      setNewCount(prev => prev + 1)
    }
  ).subscribe()
Show banner: 'N new signals — click to refresh' when newCount > 0
 
EMPTY STATES:
  No trackers: 'Create your first tracker' → /trackers
  Has trackers, no signals: 'Signals appear within 15 minutes of ingestion'

5. FEEDBACK HANDLER — in DashboardPage.tsx:
onFeedback(signalId, feedbackType):
  supabase.from('human_feedback_loop').insert({
    signal_id: signalId,
    user_id: profile.id,
    workspace_id: workspace.id,
    feedback_type: feedbackType
  })
  If feedbackType is 'not_useful':
    supabase.from('intent_signals').update({ dismissed: true })
      .eq('id', signalId)
